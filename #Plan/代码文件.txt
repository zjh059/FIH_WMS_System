        public bool InStock(string goodsCode, int qty, string locCode)
        {
            if (qty <= 0) return false;

            using (var db = new SqlConnection(connStr))
            {
                db.Open();
                // 开启“事务”防护罩：接下来的 SQL 要么全成功，要么全不作数！
                using (var transaction = db.BeginTransaction())
                {
                    try
                    {
                        // 1. 检查库存记录是不是已经存在
                        var stockId = db.ExecuteScalar<int?>(
                            "SELECT Id FROM Stock WHERE GoodsCode = @gCode AND LocationCode = @lCode",
                            new { gCode = goodsCode, lCode = locCode }, transaction);

                        if (stockId.HasValue)
                        {
                            // 如果存在，直接累加数量
                            db.Execute("UPDATE Stock SET Qty = Qty + @qty WHERE Id = @id",
                                new { qty = qty, id = stockId.Value }, transaction);
                        }
                        else
                        {
                            // 如果不存在，新增一条库存，并把库位状态改为“已被占用(1)”
                            db.Execute("INSERT INTO Stock (GoodsCode, LocationCode, Qty) VALUES (@gCode, @lCode, @qty)",
                                new { gCode = goodsCode, lCode = locCode, qty = qty }, transaction);

                            db.Execute("UPDATE Location SET IsUsed = 1 WHERE Code = @lCode",
                                new { lCode = locCode }, transaction);
                        }

                        // 2. 无论如何，都要记一笔入库流水账 (RecordType: 0 代表入库)
                        db.Execute(@"INSERT INTO StockRecord (RecordType, GoodsCode, LocationCode, Qty, OperateTime, Operator) 
                                     VALUES (0, @gCode, @lCode, @qty, GETDATE(), 'admin')",
                                     new { gCode = goodsCode, lCode = locCode, qty = qty }, transaction);

                        // 3. Over，提交保存
                        transaction.Commit();
                        return true;
                    }
                    catch (Exception)
                    {
                        // 一旦出任何错，立刻时光倒流（回滚），保证数据不乱
                        transaction.Rollback();
                        return false;
                    }
                }
            }
        }

        // 3. 出库 (真·数据库事务版)
        public bool OutStock(string goodsCode, int qty, string locCode)
        {
            if (qty <= 0) return false;

            using (var db = new SqlConnection(connStr))
            {
                db.Open();
                using (var transaction = db.BeginTransaction())
                {
                    try
                    {
                        // 1. 先去数据库查一查，这个库位上到底有没有这个商品？库存数量够不够扣？
                        var stock = db.QueryFirstOrDefault<Stock>(
                            "SELECT Id, Qty FROM Stock WHERE GoodsCode = @gCode AND LocationCode = @lCode",
                            new { gCode = goodsCode, lCode = locCode }, transaction);

                        if (stock == null || stock.Qty < qty)
                        {
                            return false; // 库存不足，或者根本没这个货，直接拒绝出库！
                        }

                        // 2. 开始扣减库存
                        if (stock.Qty == qty)
                        {
                            // 如果刚好全部出完，就把这行库存记录删掉，并把库位状态改回“未占用(0)”
                            db.Execute("DELETE FROM Stock WHERE Id = @id", new { id = stock.Id }, transaction);
                            db.Execute("UPDATE Location SET IsUsed = 0 WHERE Code = @lCode", new { lCode = locCode }, transaction);
                        }
                        else
                        {
                            // 如果只是出了一部分，就只更新剩下的数量
                            db.Execute("UPDATE Stock SET Qty = Qty - @qty WHERE Id = @id", new { qty = qty, id = stock.Id }, transaction);
                        }

                        // 3. 记一笔出库流水账 (RecordType: 1 代表出库)
                        db.Execute(@"INSERT INTO StockRecord (RecordType, GoodsCode, LocationCode, Qty, OperateTime, Operator) 
                                     VALUES (1, @gCode, @lCode, @qty, GETDATE(), 'admin')",
                                     new { gCode = goodsCode, lCode = locCode, qty = qty }, transaction);

                        // 4. 完美通关，提交保存！
                        transaction.Commit();
                        return true;
                    }
                    catch (Exception)
                    {
                        // 出任何错，时光倒流
                        transaction.Rollback();
                        return false;
                    }
                }
            }
        }




        // 4. 查询库存
        public List<Stock> GetStockList()
        {
            //return _stocks;


            // 4. 查询库存 (真正的 SQL 数据库查询版)

            // using 会帮我们在用完数据库后自动关门，极其安全
            using(var db = new SqlConnection(connStr))
            {
                // 写一段带 JOIN 的 SQL 语句，把三张表的数据一口气查出来
                string sql = @"
                    SELECT s.Id, s.Qty,
                            g.Id, g.Code, g.Name, g.Spec, g.Price,
                            l.Id, l.Code, l.Area, l.IsUsed
                    FROM Stock s
                    INNER JOIN Goods g ON s.GoodsCode = g.Code
                    INNER JOIN Location l ON s.LocationCode = l.code";

                // Dapper 的高级多表映射方法：自动把查到的 SQL 数据塞进我们定义的三个模型里
                var list = db.Query<Stock, Goods, Location, Stock>(
                    sql,
                    (stock,goods,location) =>
                    {
                        stock.Goods = goods;
                        stock.Location = location;
                        return stock;
                    },
                    splitOn: "Id,Id"
                ).ToList();

                return list;
            }
        }

        // 5. 查询出入库记录 (数据库)
        public List<StockRecord> GetRecords()
        {
            using (var db = new SqlConnection(connStr))
            {
                // 连表查询，把商品的中文名也带出来，并且按时间倒序排列
                string sql = @"
                    SELECT r.Id, r.RecordType as Type, r.Qty, r.OperateTime, r.Operator,
                           g.Id, g.Code, g.Name, g.Spec, g.Price,
                           l.Id, l.Code, l.Area, l.IsUsed
                    FROM StockRecord r
                    INNER JOIN Goods g ON r.GoodsCode = g.Code
                    INNER JOIN Location l ON r.LocationCode = l.Code
                    ORDER BY r.OperateTime DESC";

                // Dapper 自动映射方法
                var list = db.Query<StockRecord, Goods, Location, StockRecord>(
                    sql,
                    (record, goods, location) =>
                    {
                        record.Goods = goods;
                        record.Location = location;
                        return record;
                    },
                    splitOn: "Id,Id"
                ).ToList();

                return list;
            }
        }

    /// 入库窗体界面
    /// </summary>
    public partial class OutStockForm : Form
    {
        public OutStockForm()
        {
            InitializeComponent();
        }


        // 1. 定义三个“公开的口袋”，用来装用户输入的数据，方便主窗口一会来拿
        public string InputGoodsCode { get; set; }
        public int InputQty { get; set; }
        public string InputLocCode { get; set; }


        // 确认按钮的点击事件
        private void btnConfirm_Click(object sender, EventArgs e)
        {
            try
            {
                // 2. 把文本框里的字，装进我们的口袋里
                InputGoodsCode = txtGoodsCode.Text;
                InputLocCode = txtLocCode.Text;

                // 数量必须是数字，所以用 int.Parse 转换一下
                InputQty = int.Parse(txtQty.Text);

                // 3. 极其重要的一句！告诉系统：“这个弹窗的任务圆满完成 (OK)！”
                // 这句话执行后，弹窗会自动关闭，并给主窗口发个 OK 的信号。
                this.DialogResult = DialogResult.OK;
            }
            catch (Exception)
            {
                MessageBox.Show("警告：出库数量必须填入纯数字！", "输入错误", MessageBoxButtons.OK, MessageBoxIcon.Warning);
            }
        }
    }
